1. Total Registrations per Event

SELECT e.id, e.title, COUNT(r.id) AS registrations
FROM events e
LEFT JOIN registrations r ON r.event_id = e.id
WHERE e.college_id = 1
GROUP BY e.id, e.title
ORDER BY registrations DESC;
